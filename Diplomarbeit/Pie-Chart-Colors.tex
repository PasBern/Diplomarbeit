\documentclass{article}

\usepackage{pstricks-add}
\usepackage[
  round-mode = places,
  round-precision = 0
]{siunitx}

\ExplSyntaxOn
  \cs_new_eq:NN \calc \fp_eval:n
\ExplSyntaxOff

\newcommand*\total{\calc{\honey+\pollen+\water}}
\newcommand*\Honey{\calc{\honey/\total*100}}
\newcommand*\Pollen{\calc{\pollen/\total*100}}
\newcommand*\Water{\calc{\water/\total*100}}

\def\pieSlice[#1](#2)#3#4#5{%
  \rput(psChartI#2){\shortstack[c]{%
      \SI{#3}{\g}\strut\\[-1ex]
      \SI{#4}{\percent}\strut}}
  \ncline{psChartO#2}{psChart#2}
  \nput{#1}{psChartO#2}{#5}}

% parameters
\def\honey{1671}
\def\pollen{1015}
\def\water{621}

\begin{document}

\psset{nodesepB = -5pt}
\begin{pspicture}(-3,-3.5)(4.1,4.1)
  \psChart[
    userColor = {red!60, green!60, blue!60},
    chartNodeO = 1.2,
    shadow = true,
    shadowsize = 5pt,
    linewidth = 0
  ]{\honey,\pollen,\water}{}{3}
  \pieSlice[90](1){\honey}{\Honey}{Honey}
  \pieSlice[-90](2){\pollen}{\Pollen}{Pollen}
  \pieSlice[0](3){\water}{\Water}{Water}
\end{pspicture}

\end{document}
