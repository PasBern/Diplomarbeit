\subsection*{Regionale Zusammenarbeit in Europa}

\begin{itemize}
	\item \textbf{Antriebskräfte regionaler Kooperation unterhalb der inter-gouvernementalen Ebene}


	\begin{compactitem}
		\item \textsl{Subsidiaritätsprinzip:} regionale Probleme können besser von lokalen Institutionen bewältigt werden
		\item EU-Politik war nicht auf die Bedürfnisse der territorialen Gliederung der Mitgliedsstaaten zugeschnitten
		\item bisherige Möglichkeiten der Zusammenarbeit auf europäischer Ebene nicht ausreichend
		\item durch Grenzen hervorgerufene Hindernisse sollen überwunden werden
	\end{compactitem}


	\item \textbf{Rechtliche Grundlagen der grenzüberschreitenden Zusammenarbeit}

	\begin{compactitem}
		\item \textsl{Europäische Rahmenkonvention zur grenzüberschreitenden Zusammenarbeit (Konvention von Madrid, 1980)} gibt ersten rechtlichen Rahmen
		\item Zusatzprotokoll von 1995 gab erstmals die Möglichkeit, mit Gebietskörperschaften von Nicht-EU-Mitgliedern zusammenzuarbeiten
		\item \textsl{Vertrag von Maastricht 1992} erhob das Subsidiaritätsprinzip zur allgemeingültigen Rechtsgrundlage
		\item Finanzierungsmöglichkeiten für Projekte durch Europäische Territorial Zusammenarbeit (ETZ) nach Art. 158 EGV im Rahmen der Kohäsions- \& Strukturpolitik
	\end{compactitem}


	\item \textbf{Zusammenarbeit am Beispiel der Euro-Regionen}

	\begin{itemize}

		\item \textsl{Oberrheinregion (Frankreich - Schweiz - Deutschland)}


			\begin{compactitem}
			\item institutionalisierte Zusammenarbeit seit 1977 (Bonner Abkommen): Oberrheinrat und Oberrheinkonferenz
			\item Herausforderungen: Katastrophen- \& Umweltschutz, Raumpolitik (grenzüberschreitende Infrastruktur, Pendlerverkehr)m Tourismus
			\item weitreichender Handlungsspielraum in Deutschland und der Schweiz für lokale Selbstverwaltung
			\item durch zentralistische Strukturen nur bedingt Kompetenzen für D\'{e}partements und Regionen in Frankreich
			\end{compactitem}

\pagebreak


		\item \textsl{Euro-Region Niemen (Polen - Litauen - Weißrussland  - Russische Föderation)}


			\begin{compactitem}
			\item Zusammenarbeit seit 1997 auf den Gebieten der grenzüberschreitende Infrastruktur, Tourismusentwicklung \& Umweltschutz
			\item Kooperation zwischen Gebietskörperschaften der EU und einem Nicht-Mitglied (Russland)
			\item sehr unterschiedliche Kompetenzen für die Selbstverwaltungen zwischen Russland, Litauen und Polen
			\item kaum Erfahrung in Russland, Litauen und Weißrussland mit regionaler Autonomie\\
			\ding{225} grenzüberschreitende Kooperation gestaltet sich als äußerst schwierig
			\end{compactitem}


		\end{itemize}


% % % % % % % % % % % % % % % % % % % % % % % % %


	\item \textbf{Europäischer Verbund für Territoriale Zusammenarbeit (EVTZ)}



		\begin{compactitem}
		\item Instrument der EU, um interregionale Kooperation zwischen den Mitgliedern zu fördern (\textsl{EG-Verordnung 1086/2006})
		\item EVTZ setzt sich aus Mitgliedsstaaten, Gebietskörperschaften und Einrichtungen öffentlichen Rechts zusammen
		\item Kompetenzen des jeweiligen EVTZ werden von den Mitgliedern bestimmt\\
		\ding{225} EVTZ besitzt eine eigene Rechtspersönlichkeit
		\item durch zentralistische Strukturen nur bedingt Kompetenzen für D\'{e}partements und Regionen in Frankreich
		\end{compactitem}


		\item \textsl{Euro-Region Niemen (Polen - Litauen - Weißrussland  - Russische Föderation)}


			\begin{compactitem}
			\item Zusammenarbeit seit 1997 auf den Gebieten der grenzüberschreitende Infrastruktur, Tourismusentwicklung \& Umweltschutz
			\item Kooperation zwischen Gebietskörperschaften der EU und einem Nicht-Mitglied (Russland)
			\item sehr unterschiedliche Kompetenzen für die Selbstverwaltungen zwischen Russland, Litauen und Polen
			\item kaum Erfahrung in Russland, Litauen und Weißrussland mit regionaler Autonomie\\
			\ding{225} grenzüberschreitende Kooperation gestaltet sich als äußerst schwierig
			\end{compactitem}



\end{itemize}
