\subsection*{Ausgangssitution in der Ukraine}

\begin{itemize}
	\item \textbf{multi-ethnischer Staat ohne Tradition von Eigenstaatlichkeit, Demokratie und Marktwirtschaft}


	\begin{compactitem}
		\item Aufgabe des \textsl{Nation-Building}, jedoch kein nationaler Konsens zu Wirtschaftssystem\& Gesellschaftsmodell
		\item unterschiedliche Identitäten und politische Ausrichtung der ethnischen Gruppen in der Ukraine
		\item hohe Erwartungen an Wohlstandsentwicklung
	\end{compactitem}


	\item \textbf{dysfunktionales politisches System}

	\begin{compactitem}
		\item Kompetenzstreitigkeiten zwischen Präsident und Parlament
		\item Ziele der Elite sind im Widerspruch mit den Erfordernissen des Transformationsprozesses (Einführung vom Wettbewerb, Rechtsstaatlichkeit)
		\item undurchsichtige Privatisierungen, \textsl{State-Capture}, \textsl{Rent-Seeking} durch Oligarchen
		\item Interessen von Oligarchen und Machtkämpfe zwischen ihnen bestimmen ukrainische Politik
	\end{compactitem}


	\item \textbf{kontinuierliche Wirtschaftskrise der Ukraine}

	\begin{compactitem}
	\item Absatzmärkte für ukrainische Wirtschaft brachen weg (Landwirtschaft, Bergbau, Rüstungsgüter)
	\item Strukturwandel der sowjetischen Wirtschaft wurde von der Politik nicht vorangetrieben
	\item Lebensstandard der russischen Minderheit basierte auf sowjetischem System und war durch Marktreformen bedroht
	\item Schattenwirtschaft florierte als einzige Branche
	\item sinkender Lebensstandard für die Mehrheit (Entlassungen, Inflation) bei zeitgleicher Bereicherung einer Minderheit
	\end{compactitem}

\end{itemize}


\pagebreak



\begin{quote}
\textsf{\textbf{These II:}\\
		\textit{Der Status quo ante der ukrainischen Grenzen wird nicht wieder herzustellen sein, da weder die russische Minderheit, noch Russland hieran ein Interesse haben und dem Westen für entsprechende Schritte der Wille und die Einigkeit zu gemeinsamen Handeln fehlen.}}
\end{quote}

\vspace{0.8cm}


\begin{itemize}
	\item \textbf{Scheitern von Demokratie und Marktwirtschaft in der Ukraine}
	
	\begin{itemize}
	\item der Staat Ukraine hat die Erwartungen der Bevölkerung enttäuscht\\
	\ding{225} kein Vertrauen in staatliche Institutionen\\
	\ding{225} der Staat kann den Erwartungen von 'Output-Legitimation' der russischen Minderheit nicht gerecht werden
	\end{itemize}


	\item \textbf{Präsident Vladimir Putins Großmachtsambitionen}

	\begin{compactitem}
	\item Russland will alte Hegemonialstellung wiedererlangen
	\item Instabile Ukraine im Interesse russischer Außenpolitik
	\item Ukrainische Demokratie als abschreckendes Beispiel für eigene Bevölkerung
	\end{compactitem}


	\item \textbf{Unentschlossenheit des Westens (EU, NATO, USA)}

	\begin{compactitem}
	\item Garantieerklärungen Großbritanniens und der USA an die Ukraine nach Budapester Abkommen nicht eingehalten
	\item Uneinigkeit der EU-Mitglieder über Vorgehen gegenüber Russland
	\item Interessenskonflikte zwischen Geschäftsbeziehungen mit Russland und der Versorgung mit russischem Erdgas einerseits sowie dem Aufbau der Ukraine als Modell für Demokratie, Rechtsstaatlichkeit und Marktwirtschaft andererseits
	\item Ukraine als Mitglied für EU und NATO unattraktiv
	\end{compactitem}


\end{itemize}
